46/**
47 * @brief Formatea uint128_t imitando iostream con manipuladores
48 */
49inline std::string format_like_iostream(const uint128_t& value, std::ios_base::fmtflags flags,
50 std::streamsize width = 0, char fill = ' ')
51{
52
53 const int base_flags = (flags & std::ios_base::basefield);
54 const bool show_base = (flags & std::ios_base::showbase);
55 const bool uppercase = (flags & std::ios_base::uppercase);
56 const auto alignment = (flags & std::ios_base::adjustfield);
57
58 std::string result;
59
60 // Determinar la base numérica
61 switch (base_flags) {
62 case std::ios_base::hex:
63 result = value.to_string_hex(show_base);
64 if (uppercase) {
65 // Convertir a mayúsculas
66 for (char& c : result) {
67 if (c >= 'a' && c <= 'f') {
68 c = c - 'a' + 'A';
69 }
70 }
71 }
72 break;
73
74 case std::ios_base::oct:
75 result = value.to_string_oct(show_base);
76 break;
77
78 default: // std::ios_base::dec o sin especificar
79 result = value.to_string();
80 break;
81 }
82
83 // Aplicar ancho de campo y alineación
84 if (width > 0 && static_cast<int>(result.length()) < width) {
85 const int padding = width - static_cast<int>(result.length());
86
87 switch (alignment) {
88 case std::ios_base::left:
89 result += std::string(padding, fill);
90 break;
91
92 case std::ios_base::internal:
93 // Para internal, el relleno va después del prefijo pero antes del número
94 if ((show_base &&
95 (base_flags == std::ios_base::hex || base_flags == std::ios_base::oct))) {
96 if (base_flags == std::ios_base::hex && result.substr(0, 2) == "0x") {
97 result = "0x" + std::string(padding, fill) + result.substr(2);
98 } else if (base_flags == std::ios_base::hex && result.substr(0, 2) == "0X") {
99 result = "0X" + std::string(padding, fill) + result.substr(2);
100 } else if (base_flags == std::ios_base::oct && result.substr(0, 1) == "0" &&
101 result.length() > 1) {
102 result = "0" + std::string(padding, fill) + result.substr(1);
103 } else {
104 result = std::string(padding, fill) + result; // fallback a right
105 }
106 } else {
107 result = std::string(padding, fill) + result; // Sin prefijo, actúa como right
108 }
109 break;
110
111 default: // std::ios_base::right o sin especificar
112 result = std::string(padding, fill) + result;
113 break;
114 }
115 }
116
117 return result;
118}

120/**
121 * @brief Formatea uint128_t con especificaciones completas
122 */
123inline std::string format(const uint128_t& value, int base = 10, int width = 0, char fill = ' ',
124 bool show_base = false, bool uppercase = false, bool left_align = false,
125 bool internal_align = false)
126{
127
128 std::ios_base::fmtflags flags = std::ios_base::dec;
129
130 // Configurar base
131 switch (base) {
132 case 16:
133 flags = std::ios_base::hex;
134 break;
135 case 8:
136 flags = std::ios_base::oct;
137 break;
138 default:
139 flags = std::ios_base::dec;
140 break;
141 }
142
143 // Configurar opciones
144 if (show_base)
145 flags |= std::ios_base::showbase;
146 if (uppercase)
147 flags |= std::ios_base::uppercase;
148
149 // Configurar alineación
150 if (left_align) {
151 flags |= std::ios_base::left;
152 } else if (internal_align) {
153 flags |= std::ios_base::internal;
154 } else {
155 flags |= std::ios_base::right;
156 }
157
158 return format_like_iostream(value, flags, width, fill);
159}

161/**
162 * @brief Formateo hexadecimal con opciones
163 */
164inline std::string hex(const uint128_t& value, int width = 0, bool show_base = false,
165 bool uppercase = false, char fill = '0')
166{
167 return format(value, 16, width, fill, show_base, uppercase, false, width > 0);
168}
169
170/**
171 * @brief Formateo octal con opciones
172 */
173inline std::string oct(const uint128_t& value, int width = 0, bool show_base = false,
174 char fill = '0')
175{
176 return format(value, 8, width, fill, show_base, false, false, width > 0);
177}
178
179/**
180 * @brief Formateo decimal con opciones
181 */
182inline std::string dec(const uint128_t& value, int width = 0, char fill = ' ',
183 bool left_align = false)
184{
185 return format(value, 10, width, fill, false, false, left_align, false);
186}
