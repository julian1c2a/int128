724 /**
725 * @brief Realiza un desplazamiento lógico a la izquierda (left shift).
726 * @param positions El número de posiciones a desplazar.
727 * @pre `positions` debe ser un entero.
728 * @post Si `positions` es <= 0, devuelve `*this`. Si es >= 128, devuelve 0.
729 * @return Un nuevo `uint128_t` con el resultado del desplazamiento.
730 * @property Es `constexpr` y `noexcept`.
731 * @test (test_shift_left)
732 * @endcode
733 */
734 constexpr uint128_t shift_left(int positions) const noexcept
735 {
736 if (positions <= 0)
737 return *this;
738 if (positions >= 128)
739 return uint128_t(0, 0);
740
741 if (positions >= 64) {
742 return uint128_t(data[0] << (positions - 64), 0);
743 } else {
744 const uint64_t new_high = (data[1] << positions) | (data[0] >> (64 - positions));
745 const uint64_t new_low = data[0] << positions;
746 return uint128_t(new_high, new_low);
747 }
748 }

754 * @post Si `positions` es <= 0, devuelve `*this`. Si es >= 128, devuelve 0.
755 * @return Un nuevo `uint128_t` con el resultado del desplazamiento.
756 * @property Es `constexpr` y `noexcept`.
757 * @test (test_shift_right)
758 */
759 constexpr uint128_t shift_right(int positions) const noexcept
760 {
761 if (positions <= 0) {
762 return *this;
763 }
764 if (positions >= 128) {
765 return uint128_t(0, 0);
766 }
767 if (positions >= 64) {
768 return uint128_t(0, data[1] >> (positions - 64));
769 } else {
770 const uint64_t new_low = (data[0] >> positions) | (data[1] << (64 - positions));
771 const uint64_t new_high = data[1] >> positions;
772 return uint128_t(new_high, new_low);
773 }

872 /**
873 * @brief Calcula simultáneamente el cociente y el resto de una división.
874 * @details Esta es la función base para los operadores `/` y `%`. Implementa un
875 * algoritmo de división larga binaria y está optimizada para casos especiales como la
876 * división por una potencia de 2.
877 * @param divisor El valor `uint128_t` por el cual dividir.
878 * @pre El `divisor` es un `uint128_t`.
879 * @post Si el divisor es 0, el resultado es `std::nullopt`. En otro caso, el par
880 * contiene el cociente y el resto.
881 * @return Un `std::optional` que contiene un `std::pair` con el cociente (`first`) y el
882 * resto
883 * (`second`). El `optional` está vacío si ocurre una división por cero.
884 * @property Es `constexpr` y `noexcept`.
885 * @test (test_divrem_basic)
886 * @code{.cpp}
887 * // // Tanto para esta método como para los operadores / y %
888 * // // como para divisores de todos los tipos integrales soportados
889 * // // // casos especiales: división por 0, división por 1, división por potencia de
890 * 2,
891 * // // // dividendos menores que el divisor, dividendos iguales al divisor
892 * // // // casos generales: varios dividendos y divisores aleatorios
893 * // // // verificar que (quotient * divisor + remainder) == dividend
894 * @endcode
895 */
896 constexpr std::optional<std::pair<uint128_t, uint128_t>>
897 divrem(const uint128_t& divisor) const noexcept
898 {
899 // Detectar división por cero
900 if (divisor == uint128_t(0, 0)) {
901 return std::nullopt;
902 }
903
904 uint128_t quotient(0, 0);
905 uint128_t remainder(0, 0);
906
907 // Casos especiales
908 if (*this < divisor) {
909 remainder = *this;
910 return std::make_pair(quotient, remainder);
911 } else if (*this == divisor) {
912 quotient = uint128_t(0, 1);
913 return std::make_pair(quotient, remainder);
914 } else if (divisor == uint128_t(0, 1)) {
915 quotient = *this;
916 return std::make_pair(quotient, remainder);
917 } else if (divisor.is_power_of_2()) {
918 // División por potencia de 2
919 const int shift_amount = divisor.trailing_zeros();
920 quotient = this->shift_right(shift_amount);
921 remainder = *this & (divisor - uint128_t(0, 1));
922 return std::make_pair(quotient, remainder);
923 }
924
925 // CASO GENERAL: Algoritmo de división larga binaria
926 const auto normalized_pair = divisor.normalize_divisor(*this);
927 uint128_t normalized_divisor = normalized_pair.first;
928 const int shift_amount = normalized_pair.second;
929
930 remainder = *this;
931 quotient = uint128_t(0, 0);
932
933 // Bucle principal de división binaria
934 for (int i = shift_amount; i >= 0; i--) {
935 quotient = quotient.shift_left(1);
936
937 if (remainder >= normalized_divisor) {
938 remainder = remainder - normalized_divisor;
939 quotient.data[0] |= 1; // Establecer bit menos significativo
940 }
941
942 normalized_divisor = normalized_divisor.shift_right(1);
943 }
944
945 return std::make_pair(quotient, remainder);
946 }

1333 * @param other El valor `uint128_t` por el que multiplicar.
1334 * @post El valor de `*this` se actualiza con los 128 bits inferiores del producto.
1335 * @property El desbordamiento (overflow) es silencioso. Es `constexpr` y `noexcept`.
1336 * @return Una referencia a `*this`.
1337 * @test (test_mult_assignment_operator)
1338 * @code{.cpp}
1339 * // uint128_t val1(rng(), rng());
1340 * // uint128_t val2(rng(), rng());

1342 * // uint128_t val2' = val2;

1344 * // val2' *= val1;

1346 * // elemento neutro
1347 * // elemento nulo
1348 * // val2 * (val1-1) == val2 * val1 - val2
1349 * // val1 * (val2-1) == val1 * val2 - val1
1350 * // val2 * (val1+1) == val2 * val1 + val2
1351 * // val1 * (val2+1) == val1 * val2 + val1
1352 * // (val2 * val1).divrem(val1) == [val2,0]
1353 * // (val2 * val1).divrem(val2) == [val1,0]
1354 * @endcode
1355 */
1356 constexpr uint128_t& operator*=(const uint128_t& other) noexcept
1357 {
1358#if defined(__SIZEOF_INT128__)
1359 // Ruta optimizada para compiladores con soporte nativo de 128 bits (GCC/Clang)
1360 __uint128_t lhs = (static_cast<__uint128_t>(data[1]) << 64) | data[0];
1361 __uint128_t rhs = (static_cast<__uint128_t>(other.data[1]) << 64) | other.data[0];
1362 __uint128_t res = lhs * rhs;
1363 data[0] = static_cast<uint64_t>(res);
1364 data[1] = static_cast<uint64_t>(res >> 64);
1365#else
1366 // Fallback manual compatible con constexpr (para MSVC y otros)
1367 const uint64_t lhs_lo = data[0];
1368 const uint64_t lhs_hi = data[1];
1369 const uint64_t rhs_lo = other.data[0];
1370 const uint64_t rhs_hi = other.data[1];
1371
1372 // Multiplicación 64x64 -> 128 (parte baja)
1373 const uint64_t u1 = lhs_lo & 0xffffffff;
1374 const uint64_t v1 = rhs_lo & 0xffffffff;
1375 uint64_t t = u1 * v1;
1376 const uint64_t w3 = t & 0xffffffff;
1377 uint64_t k = t >> 32;
1378
1379 const uint64_t u0 = lhs_lo >> 32;
1380 t = (u0 * v1) + k;
1381 k = t & 0xffffffff;
1382 const uint64_t w1 = t >> 32;
1383
1384 const uint64_t v0 = rhs_lo >> 32;
1385 t = (u1 * v0) + k;
1386 k = t >> 32;
1387
1388 const uint64_t lo_hi = (u0 * v0) + w1 + k;
1389 const uint64_t lo_lo = (t << 32) + w3;
1390
1391 // Resultado final
1392 data[0] = lo_lo;
1393 data[1] = (lhs_hi * rhs_lo) + (lhs_lo * rhs_hi) + lo_hi;
1394#endif

1396 }

1549 /// Esta función calcula los bits superiores del resultado de multiplicar
1550 /// un número de 128 bits por uno de 64 bits, útil para el algoritmo de Knuth
1551
1552 private:
1553 // Emulación optimizada de __umulh para GCC/Clang
1554 static inline uint64_t umulh_emulation(uint64_t a, uint64_t b) noexcept
1555 {
1556#if defined(__x86_64__) && defined(__BMI2__)
1557 // BMI2: usar _mulx_u64 (equivalente directo a __umulh)
1558 unsigned long long high_part;
1559 (void)_mulx_u64(a, b, &high_part);
1560 return high_part;
1561
1562#elif defined(__SIZEOF_INT128__)
1563 // __uint128_t: más portable y generalmente muy eficiente
1564 const __uint128_t result = static_cast<__uint128_t>(a) * b;
1565 return static_cast<uint64_t>(result >> 64);
1566
1567#elif defined(__x86_64__) && (defined(__GNUC__) || defined(__clang__))
1568 // Inline assembly: mapeo directo a mulq (exactamente como __umulh)
1569 uint64_t high_part, low_part;
1570 __asm__("mulq %3" : "=a"(low_part), "=d"(high_part) : "a"(a), "r"(b) : "cc");
1571 return high_part;
1572
1573#else
1574 // Fallback: implementación manual optimizada
1575 const uint32_t a_lo = static_cast<uint32_t>(a);
1576 const uint32_t a_hi = static_cast<uint32_t>(a >> 32);
1577 const uint32_t b_lo = static_cast<uint32_t>(b);
1578 const uint32_t b_hi = static_cast<uint32_t>(b >> 32);
1579
1580 const uint64_t p0 = static_cast<uint64_t>(a_lo) * b_lo;
1581 const uint64_t p1 = static_cast<uint64_t>(a_lo) * b_hi;
1582 const uint64_t p2 = static_cast<uint64_t>(a_hi) * b_lo;
1583 const uint64_t p3 = static_cast<uint64_t>(a_hi) * b_hi;
1584
1585 const uint64_t middle = p1 + (p0 >> 32) + static_cast<uint32_t>(p2);
1586
1587 return p3 + (middle >> 32) + (p2 >> 32);
1588#endif

1763 const uint64_t carry = (sum < low_high) ? 1 : 0;

1765 return high_high + carry;

1769 /**
1770 * @brief División avanzada de 128 bits usando el Algoritmo D de Knuth con
1771 * optimizaciones.
1772 * @details Implementa una división precisa de 128 por 128 bits. Incluye múltiples rutas
1773 * rápidas para casos comunes:
1774 * - División por cero.
1775 * - Divisor mayor que dividendo.
1776 * - Divisor es potencia de 2.
1777 * - Divisor es potencia de 10, 3 o 5.
1778 * - Ambos operandos caben en 64 bits.
1779 * - Divisor cabe en 64 bits (usando `__uint128_t` nativo si está disponible).
1780 * Si ninguna optimización aplica, recurre al Algoritmo D completo.
1781 * @param v_in El divisor de 128 bits.
1782 * @pre `v_in` es un `uint128_t`.
1783 * @post No modifica el estado de `*this`.
1784 * @return Un `std::optional<std::pair<uint128_t, uint128_t>>` con `{cociente, resto}`.
1785 * Si `v_in` es 0, devuelve `std::nullopt`.
1786 * @property Es `noexcept`.
1787 * @test (test_knuth_D_divrem)
1788 */
1789 std::optional<std::pair<uint128_t, uint128_t>>
1790 knuth_D_divrem(const uint128_t& v_in) const noexcept
1791 {
1792 // 0. Casos triviales
1793 if (v_in == uint128_t(0, 0))
1794 return std::nullopt; // División por cero
1795 if (*this < v_in) {
1796 return std::make_pair(uint128_t(0, 0), *this);
1797 }
1798
1799 // === OPTIMIZACIONES RÁPIDAS ===
1800
1801 // 1. OPTIMIZACIÓN: División por potencias de 2 (usar shift)
1802 if (v_in.is_power_of_2()) {
1803 const int shift_amount = count_trailing_zeros(v_in);
1804 const uint128_t quotient = this->shift_right(shift_amount);
1805 const uint128_t mask = v_in - uint128_t(0, 1); // 2^n - 1
1806 const uint128_t remainder = *this & mask;
1807 return std::make_pair(quotient, remainder);
1808 }
1809
1810 // 2. OPTIMIZACIÓN: División por potencias de 10 (descomponer en 2^n * 5^n)
1811 auto power_of_10_result = try_divide_by_power_of_10(v_in);
1812 if (power_of_10_result.has_value()) {
1813 return power_of_10_result;
1814 }
1815
1816 // 3. OPTIMIZACIÓN: División por potencias de 3
1817 auto power_of_3_result = try_divide_by_power_of_3(v_in);
1818 if (power_of_3_result.has_value()) {
1819 return power_of_3_result;
1820 }
1821
1822 // 4. OPTIMIZACIÓN: División por potencias de 5
1823 auto power_of_5_result = try_divide_by_power_of_5(v_in);
1824 if (power_of_5_result.has_value()) {
1825 return power_of_5_result;
1826 }
1827
1828 // 5. OPTIMIZACIÓN: Ambos números caben efectivamente en 64 bits
1829 if (fits_in_64_bits(*this) && fits_in_64_bits(v_in)) {
1830 const uint64_t dividend_64 = effective_to_64_bits(*this);
1831 const uint64_t divisor_64 = effective_to_64_bits(v_in);
1832 const uint64_t q_64 = dividend_64 / divisor_64;
1833 const uint64_t r_64 = dividend_64 % divisor_64;
1834 return std::make_pair(uint128_t(0, q_64), uint128_t(0, r_64));
1835 }
1836
1837 // === RUTAS ESTÁNDAR ===
1838
1839 // Si el divisor cabe en 64 bits (data[1] == 0), usamos una ruta rápida.
1840 if (v_in.data[1] == 0) {
1841#if defined(__SIZEOF_INT128__) && !(defined(__INTEL_LLVM_COMPILER) && defined(_MSC_VER))
1842 // Intel ICX en Windows usa el linker de MSVC que no proporciona __udivti3/__umodti3
1843 const __uint128_t dividend = (static_cast<__uint128_t>(data[1]) << 64) | data[0];
1844 const uint64_t divisor = v_in.data[0];
1845 const __uint128_t q = dividend / divisor;
1846 const __uint128_t r = dividend % divisor;
1847 const uint128_t quotient(static_cast<uint64_t>(q >> 64), static_cast<uint64_t>(q));
1848 const uint128_t remainder(0, static_cast<uint64_t>(r));
1849 return std::make_pair(quotient, remainder);
1850#else
1851 // Fallback para compiladores sin __uint128_t o Intel ICX en Windows
1852 return divrem(v_in);
1853#endif
1854 }
1855
1856 // --- ALGORITMO D DE KNUTH (Para divisor > 64 bits) ---
1857
1858 // D1. Normalización
1859 // Desplazamos u y v para que el MSB de v sea 1.
1860 const int s = v_in.leading_zeros() > 0 ? v_in.leading_zeros() - 64 : 0;
1861 [[maybe_unused]] const uint128_t v = v_in.shift_left(s);
1862 [[maybe_unused]] const uint128_t u_shifted = this->shift_left(s);
1863
1864 // Capturamos el dígito extra de u que se salió por la izquierda al hacer shift.
1865 [[maybe_unused]] uint64_t u_extension = 0;
1866 if (s > 0) {
1867 u_extension = data[1] >> (64 - s);
1868 }
1869
1870 // D3. Calcular q_hat (Estimación del cociente)
1871 // Dividimos los dos dígitos más significativos del dividendo
1872#if defined(__SIZEOF_INT128__) && !(defined(__INTEL_LLVM_COMPILER) && defined(_MSC_VER))
1873 // Intel ICX en Windows usa el linker de MSVC que no proporciona __udivti3/__umodti3
1874 const __uint128_t numerator =
1875 (static_cast<__uint128_t>(u_extension) << 64) | u_shifted.data[1];
1876 const uint64_t divisor_high = v.data[1];
1877
1878 // División nativa de 128/64 para estimar
1879 __uint128_t q_hat_wide = numerator / divisor_high;
1880 __uint128_t r_hat_wide = numerator % divisor_high;
1881
1882 uint64_t q_hat = static_cast<uint64_t>(q_hat_wide);
1883
1884 // Ajuste de q_hat: El bucle interno de Knuth para refinar la estimación
1885 while (true) {
1886 if (q_hat_wide >= 0xFFFFFFFFFFFFFFFFULL) {
1887 q_hat--;
1888 r_hat_wide += divisor_high;
1889 if (r_hat_wide > 0xFFFFFFFFFFFFFFFFULL)
1890 continue;
1891 }
1892
1893 const __uint128_t left = static_cast<__uint128_t>(q_hat) * v.data[0];
1894 const __uint128_t right = (r_hat_wide << 64) | u_shifted.data[0];
1895
1896 if (left > right) {
1897 q_hat--;
1898 r_hat_wide += divisor_high;
1899 if (r_hat_wide > 0xFFFFFFFFFFFFFFFFULL)
1900 break;
1901 } else {
1902 break;
1903 }
1904 q_hat_wide = q_hat;
1905 }
1906
1907 // D4. Multiplicar y Restar: u -= q_hat * v
1908 const uint128_t p_low = v * uint128_t(0, q_hat);
1909 const uint64_t p_ext = v.fullmult_times_uint64(q_hat);
1910
1911 // Guardamos estado para detectar "borrow" (underflow) en la resta de 128 bits
1912 const uint128_t u_shifted_old = u_shifted;
1913 uint128_t u_result = u_shifted - p_low;
1914
1915 // Detectamos si la resta pidió prestado
1916 const uint64_t borrow = (u_result > u_shifted_old) ? 1 : 0;
1917
1918 // Realizamos la resta final en la parte más significativa
1919 const int64_t final_balance =
1920 static_cast<int64_t>(u_extension) - static_cast<int64_t>(p_ext + borrow);
1921
1922 // D5. Test de Resto (Corrección de signo)
1923 // Si el balance es negativo, q_hat era 1 unidad muy grande.
1924 if (final_balance < 0) {
1925 // D6. Add Back
1926 q_hat--;
1927 u_result += v; // Devolvemos una vez el divisor al resto
1928 }
1929
1930 const uint128_t quotient(0, q_hat);
1931
1932 // D8. Desnormalizar Resto
1933 const uint128_t remainder = u_result.shift_right(s);
1934
1935 return std::make_pair(quotient, remainder);
1936#else
1937 // Fallback sin __uint128_t - usar algoritmo básico
1938 return divrem(v_in);
1939#endif
1940 }

2351 * @param str La `std::string` de entrada.
2352 * @param base La base numérica (entre 2 y 36).
2353 * @return El `uint128_t` resultante. Devuelve 0 si los parámetros son inválidos.
2354 * @property Es `static`.
2355 * @test (test_from_string_base)
2356 * @code{.cpp}
2357 * // uint128_t val;
2358 * // val.set_low(0x1234);
2359 * // assert(val.low() == 0x1234);
2360 * @endcode
2361 */
2362 static uint128_t from_string_base(const std::string& str, int base)
2363 {
2364 if (str.empty() || base < 2 || base > 36) {
2365 return uint128_t(0, 0);
2366 }
2367
2368 uint128_t result(0, 0);
2369 const uint128_t base_val(0, static_cast<uint64_t>(base));
2370
2371 for (char c : str) {
2372 int digit_value = -1;
2373
2374 if (c >= '0' && c <= '9') {
2375 digit_value = c - '0';
2376 } else if (c >= 'A' && c <= 'Z') {
2377 digit_value = c - 'A' + 10;
2378 } else if (c >= 'a' && c <= 'z') {
2379 digit_value = c - 'a' + 10;
2380 }
2381
2382 if (digit_value == -1 || digit_value >= base) {
2383 // Caracteres inválidos se ignoran
2384 continue;
2385 }
2386
2387 result *= base_val;
2388 result += uint128_t(0, static_cast<uint64_t>(digit_value));
2389 }
2390
2391 return result;
2392 }
2393};
