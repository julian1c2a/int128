34/**
35 * @brief Conceptos personalizados para uint128_t
36 *
37 * Este archivo define conceptos C++20 específicos para trabajar con uint128_t
38 * y proporciona herramientas de metaprogramación avanzadas.
39 */
40
41namespace uint128_concepts
42{
43
44// ===============================================================================
45// CONCEPTOS BÁSICOS
46// ===============================================================================
47
48/**
49 * @brief Concept para tipos que pueden convertirse a uint128_t
50 *
51 * @tparam T Tipo a verificar
52 *
53 * Un tipo satisface este concepto si es un tipo integral estándar
54 * o es el mismo uint128_t.
55 */
56template <typename T>
57concept uint128_convertible = std::integral<T> || std::same_as<T, uint128_t>;
58
59/**
60 * @brief Concept para tipos que son compatibles con operaciones de uint128_t
61 *
62 * @tparam T Tipo a verificar
63 *
64 * Incluye todos los tipos convertibles a uint128_t y también tipos
65 * de punto flotante para operaciones mixtas.
66 */
67template <typename T>
68concept uint128_compatible = uint128_convertible<T> || std::floating_point<T>;
69
70/**
71 * @brief Concept para tipos que pueden usar con operaciones bitwise
72 *
73 * @tparam T Tipo a verificar
74 *
75 * Solo tipos integrales pueden realizar operaciones bitwise seguras.
76 */
77template <typename T>
78concept uint128_bitwise_compatible = std::integral<T> || std::same_as<T, uint128_t>;
79
80// ===============================================================================
81// CONCEPTOS PARA OPERACIONES ESPECÍFICAS
82// ===============================================================================
83
84/**
85 * @brief Concept para tipos válidos en operaciones de desplazamiento
86 *
87 * @tparam T Tipo del operando de desplazamiento
88 *
89 * Los operandos de desplazamiento deben ser tipos integrales pequeños
90 * para evitar comportamiento indefinido.
91 */
92template <typename T>
93concept valid_shift_type = std::integral<T> && sizeof(T) <= 8;
94
95/**
96 * @brief Concept para tipos que pueden usarse como índices de bit
97 *
98 * @tparam T Tipo del índice
99 *
100 * Los índices de bit deben ser tipos enteros no negativos.
101 */
102template <typename T>
103concept bit_index_type = std::unsigned_integral<T> && sizeof(T) <= 8;
104
105/**
106 * @brief Concept para tipos que pueden usarse en comparaciones con uint128_t
107 *
108 * @tparam T Tipo a comparar
109 *
110 * Las comparaciones son seguras con tipos integrales y uint128_t mismo.
111 */
112template <typename T>
113concept uint128_comparable = std::integral<T> || std::same_as<T, uint128_t>;
114
115// ===============================================================================
116// CONCEPTOS PARA METAPROGRAMACIÓN
117// ===============================================================================
118
119/**
120 * @brief Concept que verifica si un tipo puede ser promovido a uint128_t
121 *
122 * @tparam T Tipo a verificar
123 *
124 * Un tipo puede ser promovido si es un tipo integral más pequeño que uint128_t.
125 */
126template <typename T>
127concept promotable_to_uint128 = std::integral<T> && sizeof(T) <= 16 && !std::same_as<T, uint128_t>;
128
129/**
130 * @brief Concept para funciones que pueden operar sobre uint128_t
131 *
132 * @tparam F Tipo de función
133 * @tparam Args Tipos de argumentos
134 *
135 * Verifica que una función pueda ser llamada con uint128_t como primer argumento.
136 */
137template <typename F, typename... Args>
138concept uint128_function = requires(F f, uint128_t u, Args... args) { f(u, args...); };
139
140/**
141 * @brief Concept para tipos que pueden actuar como contenedores de uint128_t
142 *
143 * @tparam Container Tipo de contenedor
144 *
145 * Verifica que el contenedor tenga las operaciones básicas necesarias.
146 */
147template <typename Container>
148concept uint128_container = requires(Container c) {
149 typename Container::value_type;
150 requires std::same_as<typename Container::value_type, uint128_t>;
151 c.begin();
152 c.end();
153 c.size();
154};
155
156// ===============================================================================
157// CONCEPTOS PARA ALGORITMOS
158// ===============================================================================
159
160/**
161 * @brief Concept para predicados que operan sobre uint128_t
162 *
163 * @tparam Pred Tipo del predicado
164 *
165 * Un predicado debe retornar un tipo convertible a bool cuando
166 * se aplica a uint128_t.
167 */
168template <typename Pred>
169concept uint128_predicate = requires(Pred p, uint128_t u) {
170 { p(u) } -> std::convertible_to<bool>;
171};
172
173/**
174 * @brief Concept para operaciones binarias sobre uint128_t
175 *
176 * @tparam BinaryOp Tipo de la operación binaria
177 *
178 * La operación debe poder tomar dos uint128_t y retornar un resultado.
179 */
180template <typename BinaryOp>
181concept uint128_binary_operation = requires(BinaryOp op, uint128_t a, uint128_t b) { op(a, b); };
182
183/**
184 * @brief Concept para operaciones de reducción sobre uint128_t
185 *
186 * @tparam ReduceOp Tipo de la operación de reducción
187 * @tparam T Tipo del valor inicial/acumulado
188 *
189 * La operación debe poder combinar un acumulador con un uint128_t.
190 */
191template <typename ReduceOp, typename T>
192concept uint128_reduce_operation = requires(ReduceOp op, T acc, uint128_t u) {
193 { op(acc, u) } -> std::convertible_to<T>;
194};
195
196} // namespace uint128_concepts

202/**
203 * @brief Verifica que uint128_t satisfaga los conceptos estándar de C++20
204 *
205 * Estas verificaciones se ejecutan en tiempo de compilación para
206 * asegurar compatibilidad con la biblioteca estándar.
207 */
208namespace uint128_concept_checks
209{
210// Verificaciones básicas de conceptos
211// NOTA: uint128_t es una clase, no un tipo integral nativo de C++,
212// por lo que no satisface std::integral (requiere std::is_integral_v<T>)
213// pero implementa todas las operaciones de un tipo entero sin signo.
214
215// static_assert(std::integral<uint128_t>, "uint128_t debe satisfacer std::integral");
216// static_assert(std::unsigned_integral<uint128_t>,
217// "uint128_t debe satisfacer std::unsigned_integral");
218// static_assert(!std::signed_integral<uint128_t>,
219// "uint128_t no debe satisfacer std::signed_integral");
220
221static_assert(std::regular<uint128_t>, "uint128_t debe satisfacer std::regular");
222static_assert(std::totally_ordered<uint128_t>, "uint128_t debe satisfacer std::totally_ordered");
223
224// Verificaciones de operaciones
225static_assert(std::equality_comparable<uint128_t>, "uint128_t debe ser comparable por igualdad");
226// NOTA: uint128_t no implementa operator<=> (C++20 three-way comparison)
227// static_assert(std::three_way_comparable<uint128_t>,
228// "uint128_t debe soportar comparación a tres vías");
229
230// Verificaciones de construcción y asignación
231static_assert(std::default_initializable<uint128_t>, "uint128_t debe ser default-constructible");
232static_assert(std::copy_constructible<uint128_t>, "uint128_t debe ser copy-constructible");
233static_assert(std::move_constructible<uint128_t>, "uint128_t debe ser move-constructible");
234static_assert(std::assignable_from<uint128_t&, const uint128_t&>,
235 "uint128_t debe soportar asignación");
236
237} // namespace uint128_concept_checks
