53/**
54 * @brief Busqueda binaria optimizada para rangos de uint128_t
55 *
56 * @tparam ForwardIt Tipo de iterador
57 * @param first Iterador al inicio del rango
58 * @param last Iterador al final del rango
59 * @param value Valor a buscar
60 * @return true si se encuentra el valor, false en caso contrario
61 *
62 * Optimizada para aprovechar las caracteristicas especificas de uint128_t.
63 *
64 * @test test_binary_search_uint128
65 * @code{.cpp}
66 * // Verifica busqueda de valores existentes e inexistentes
67 * std::vector<uint128_t> vec = {10, 20, 30, 40, 50};
68 * assert(binary_search_uint128(vec.begin(), vec.end(), uint128_t(30)));
69 * assert(!binary_search_uint128(vec.begin(), vec.end(), uint128_t(15)));
70 * @endcode
71 */
72template <std::forward_iterator ForwardIt>
73bool binary_search_uint128(ForwardIt first, ForwardIt last, const uint128_t& value)
74 requires std::same_as<typename std::iterator_traits<ForwardIt>::value_type, uint128_t>
75{
76 first = std::lower_bound(first, last, value);
77 return (first != last) && (*first == value);
78}

304/**
305 * @brief Encuentra el maximo comun divisor de todos los elementos en un rango
306 *
307 * @tparam InputIt Tipo de iterador
308 * @param first Iterador al inicio del rango
309 * @param last Iterador al final del rango
310 * @return GCD de todos los elementos (0 si el rango esta vacio)
311 *
312 * @test test_gcd_range
313 * @code{.cpp}
314 * // Calcula GCD de multiples numeros, verifica coprimos
315 * std::vector<uint128_t> vec = {12, 18, 24};
316 * auto gcd = gcd_range(vec.begin(), vec.end());
317 * assert(gcd == uint128_t(6));
318 * @endcode
319 */
320template <std::input_iterator InputIt>
321uint128_t gcd_range(InputIt first, InputIt last)
322 requires std::same_as<typename std::iterator_traits<InputIt>::value_type, uint128_t>
323{
324 if (first == last)
325 return uint128_t(0);
326
327 uint128_t result = *first++;
328 while (first != last) {
329 result = std::gcd(result, *first++);
330 if (result == uint128_t(1))
331 break; // Optimizacion: si GCD es 1, no puede mejorar
332 }
333 return result;
334}

338 *
339 * @tparam InputIt Tipo de iterador

342 * @return LCM de todos los elementos (0 si algun elemento es 0)

344 * @test test_lcm_range

357 return uint128_t(0);

366/**
367 * @brief Genera una secuencia aritmetica de uint128_t
368 *
369 * @tparam OutputIt Tipo de iterador de salida
370 * @param first Iterador al inicio del rango de salida
371 * @param count Numero de elementos a generar
372 * @param start Valor inicial
373 * @param step Paso de la secuencia
374 *
375 * @test test_generate_arithmetic_sequence
376 * @code{.cpp}
377 * // Genera secuencias ascendentes/descendentes (10,15,20,...)
378 * generate_arithmetic_sequence(seq.begin(), 5, uint128_t(10), uint128_t(5));
379 * assert(seq[0] == uint128_t(10) && seq[4] == uint128_t(30));
380 * @endcode
381 */
382template <std::output_iterator<uint128_t> OutputIt>
383void generate_arithmetic_sequence(OutputIt first, std::size_t count, const uint128_t& start,
384 const uint128_t& step)
385{
386 uint128_t current = start;
387 for (std::size_t i = 0; i < count; ++i) {
388 *first++ = current;
389 current += step;
390 }
391}

424/**
425 * @brief Estadisticas basicas de un rango de uint128_t
426 */
427struct uint128_stats {
428 uint128_t min_value;
429 uint128_t max_value;
430 uint128_t sum;
431 std::size_t count;
432
433 double mean() const
434 {
435 return count > 0 ? static_cast<double>(static_cast<uint64_t>(sum)) / count : 0.0;
436 }
437};

439/**
440 * @brief Calcula estadisticas basicas de un rango
441 *
442 * @tparam InputIt Tipo de iterador
443 * @param first Iterador al inicio del rango
444 * @param last Iterador al final del rango
445 * @return Estructura con estadisticas basicas
446 *
447 * @test test_calculate_stats
448 * @code{.cpp}
449 * // Calcula min, max, sum, count y mean de un rango
450 * auto stats = calculate_stats(vec.begin(), vec.end());
451 * assert(stats.min_value == uint128_t(10) && stats.max_value == uint128_t(50));
452 * assert(stats.sum == uint128_t(150) && stats.mean() == 30.0);
453 * @endcode
454 */
455template <std::input_iterator InputIt>
456uint128_stats calculate_stats(InputIt first, InputIt last)
457 requires std::same_as<typename std::iterator_traits<InputIt>::value_type, uint128_t>
458{
459 if (first == last) {
460 return {uint128_t(0), uint128_t(0), uint128_t(0), 0};
461 }
462
463 uint128_stats stats;
464 stats.min_value = stats.max_value = stats.sum = *first;
465 stats.count = 1;
466
467 ++first;
468 while (first != last) {
469 const uint128_t& value = *first++;
470 stats.min_value = std::min(stats.min_value, value);
471 stats.max_value = std::max(stats.max_value, value);
472 stats.sum += value;
473 ++stats.count;
474 }
475
476 return stats;
477}
