169/**
170 * @brief Especialización de std::hash para uint128_t
171 *
172 * Proporciona una función hash que combina los valores high y low
173 * del uint128_t para su uso en containers no ordenados como
174 * std::unordered_set y std::unordered_map.
175 */
176template <> struct hash<uint128_t> {
177 /**
178 * @brief Calcula el valor hash de un uint128_t
179 * @param value Valor a hashear
180 * @return Valor hash resultante
181 *
182 * La implementación combina los hashes de las partes high y low
183 * usando XOR y desplazamiento para reducir colisiones.
184 */
185 size_t operator()(const uint128_t& value) const noexcept
186 {
187 hash<uint64_t> hasher;
188 return hasher(value.high()) ^ (hasher(value.low()) << 1);
189 }
190};
