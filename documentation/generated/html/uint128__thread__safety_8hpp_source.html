1/**
2 * @file uint128_thread_safety.hpp
3 * @brief Thread-safe wrappers for uint128_t
4 *
5 * This header provides several thread-safe wrapper classes for uint128_t:
6 *
7 * 1. ThreadSafeUint128 - Mutex-based (general purpose)
8 * 2. ThreadSafeUint128RW - Read-Write lock (optimized for read-heavy workloads)
9 * 3. ThreadSafeUint128Atomic - std::atomic wrapper (uses locks internally on most platforms)
10 * 4. ThreadSafeUint128SpinLock - Spin-lock based (low overhead for short critical sections)
11 *
12 * Performance comparison (100k operations):
13 * - Thread-local (no wrapper): ~0 µs
14 * - Mutex wrapper: ~7,000 µs
15 * - RW-lock wrapper: ~5,000 µs (read-heavy)
16 * - Atomic wrapper: ~7,000 µs
17 * - Spin-lock wrapper: ~3,000 µs
18 *
19 * Recommended usage:
20 * - For most cases: ThreadSafeUint128 (mutex-based)
21 * - For read-heavy: ThreadSafeUint128RW (RW-lock)
22 * - For thread-local: Use uint128_t directly (no wrapper needed)
23 */
24

38/**
39 * @class ThreadSafeUint128
40 * @brief Thread-safe wrapper for uint128_t using std::mutex
41 *
42 * This is the recommended general-purpose thread-safe wrapper.
43 * All operations are serialized using a mutex.
44 *
45 * Example:
46 * @code
47 * ThreadSafeUint128 counter(uint128_t(0));
48 *
49 * // From multiple threads:
50 * counter.add(uint128_t(1)); // Thread-safe increment
51 * uint128_t val = counter.get(); // Thread-safe read
52 * counter.set(uint128_t(100)); // Thread-safe write
53 * @endcode
54 *
55 * @note All operations acquire exclusive lock
56 * @note Overhead: ~70ns per operation
57 */

253 * @endcode

497/**
498 * @class ThreadSafeUint128SpinLock
499 * @brief Thread-safe wrapper using spin-lock
500 *
501 * Uses std::atomic_flag for a simple spin-lock. Best for very short
502 * critical sections where contention is low.
503 *
504 * Example:
505 * @code
506 * ThreadSafeUint128SpinLock counter(uint128_t(0));
507 *
508 * counter.add(uint128_t(1)); // Spins if locked
509 * uint128_t val = counter.get();
510 * @endcode
511 *
512 * @warning Can waste CPU cycles if contention is high
513 * @note Best for low-contention, short critical sections
514 * @note Overhead: ~30ns per operation (low contention)
515 */
516class ThreadSafeUint128SpinLock
517{
518 private:
519 uint128_t value_;
520 mutable std::atomic_flag lock_ = ATOMIC_FLAG_INIT;
521
522 void acquire() const
523 {
524 while (lock_.test_and_set(std::memory_order_acquire)) {
525 // Spin-wait
526#if defined(__x86_64__) || defined(_M_X64)
527 __builtin_ia32_pause(); // Reduce power consumption during spin
528#endif
529 }
530 }
531
532 void release() const
533 {
534 lock_.clear(std::memory_order_release);
535 }
536
537 public:
538 explicit ThreadSafeUint128SpinLock(uint128_t val = uint128_t(0)) : value_(val) {}
539
540 uint128_t get() const
541 {
542 acquire();
543 uint128_t result = value_;
544 release();
545 return result;
546 }
547
548 void set(uint128_t val)
549 {
550 acquire();
551 value_ = val;
552 release();
553 }

620/*
621 * PERFORMANCE COMPARISON (100,000 operations):
622 *
623 * Method | Time (µs) | Lock-Free | Use Case
624 * --------------------------|-----------|-----------|---------------------------
625 * Thread-local (no wrapper) | 0 | N/A | Each thread has own copy
626 * SpinLock wrapper | 3,000 | Yes | Low contention, short ops
627 * RW-Lock wrapper | 5,000 | No | Read-heavy workloads
628 * Mutex wrapper | 7,000 | No | General purpose (recommended)
629 * Atomic wrapper | 7,000 | No* | *Uses locks on most platforms
630 *
631 *
632 * RECOMMENDATIONS BY USE CASE:
633 *
634 * 1. Thread-local variables:
635 * → Use uint128_t directly (no wrapper needed)
636 * → Best performance, no overhead
637 *
638 * 2. Global shared counters:
639 * → Use ThreadSafeUint128 (mutex-based)
640 * → Simple, correct, good enough for most cases
641 *
642 * 3. Read-heavy statistics:
643 * → Use ThreadSafeUint128RW (RW-lock)
644 * → Multiple concurrent readers
645 *
646 * 4. High-frequency, low-contention:
647 * → Use ThreadSafeUint128SpinLock
648 * → Lowest overhead when contention is low
649 *
650 * 5. Standard atomic interface:
651 * → Use ThreadSafeUint128Atomic
652 * → Familiar std::atomic API
653 * → Check is_lock_free() first
654 *
655 *
656 * EXAMPLE USAGE:
657 *
658 * // Option 1: Mutex (recommended for most cases)
659 * uint128_threadsafe::ThreadSafeUint128 counter(uint128_t(0));
660 * counter.add(uint128_t(1));
661 * uint128_t value = counter.get();
662 *
663 * // Option 2: RW-Lock (for read-heavy)
664 * uint128_threadsafe::ThreadSafeUint128RW stats(uint128_t(0));
665 * uint128_t val = stats.get(); // Multiple concurrent reads OK
666 * stats.add(uint128_t(1)); // Exclusive write
667 *
668 * // Option 3: Atomic (standard interface)
669 * uint128_threadsafe::ThreadSafeUint128Atomic atomic_val(uint128_t(0));
670 * atomic_val.store(uint128_t(100));
671 * uint128_t loaded = atomic_val.load();
672 *
673 * // Option 4: SpinLock (low contention)
674 * uint128_threadsafe::ThreadSafeUint128SpinLock fast_counter(uint128_t(0));
675 * fast_counter.add(uint128_t(1));
676 *
677 *
678 * IMPORTANT NOTES:
679 *
680 * - uint128_t itself IS thread-safe for:
681 * * Concurrent reads (const operations)
682 * * Thread-local objects
683 * * Operations that return new objects
684 *
685 * - Wrappers are ONLY needed for:
686 * * Concurrent modifications of shared objects
687 * * Global/static variables accessed by multiple threads
688 *
689 * - For maximum performance:
690 * * Use thread-local pattern (each thread has own counter)
691 * * Aggregate results at the end
692 * * No synchronization overhead!
693 */
