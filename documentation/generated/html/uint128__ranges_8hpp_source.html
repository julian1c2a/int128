33/**
34 * @brief Especializaciones de funciones de rango de std para uint128_t
35 *
36 * Este archivo proporciona especializaciones de las funciones de rango
37 * estándar (std::iota, std::accumulate, etc.) para trabajar con uint128_t.
38 */
39
40namespace std
41{
42
43/**
44 * @brief Especialización de std::iota para uint128_t
45 * Llena un rango con valores secuenciales incrementales
46 */
47template <class ForwardIt> void iota(ForwardIt first, ForwardIt last, uint128_t value)
48{
49 while (first != last) {
50 *first++ = value;
51 ++value;
52 }
53}
54
55/**
56 * @brief Especialización de std::accumulate para uint128_t
57 */
58template <class InputIt> uint128_t accumulate(InputIt first, InputIt last, uint128_t init)
59{
60 while (first != last) {
61 init = init + *first;
62 ++first;
63 }
64 return init;
65}
66
67/**
68 * @brief Especialización de std::accumulate con operación personalizada
69 */
70template <class InputIt, class BinaryOperation>
71uint128_t accumulate(InputIt first, InputIt last, uint128_t init, BinaryOperation op)
72{
73 while (first != last) {
74 init = op(init, *first);
75 ++first;
76 }
77 return init;
78}
79
80/**
81 * @brief Especialización de std::inner_product para uint128_t
82 */
83template <class InputIt1, class InputIt2>
84uint128_t inner_product(InputIt1 first1, InputIt1 last1, InputIt2 first2, uint128_t init)
85{
86 while (first1 != last1) {
87 init = init + (*first1) * (*first2);
88 ++first1;
89 ++first2;
90 }
91 return init;
92}

98//
99// Solo especializamos iota() y accumulate() porque toman valores por copia
100// y podemos optimizar pasándolos directamente como uint128_t.
